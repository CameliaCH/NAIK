-- ────────────────────────────────────────────────────────────────
-- 1. USERS TABLE
--    Your existing table — only run if it doesn't exist yet.
--    The id column is bigint (Supabase default identity).
-- ────────────────────────────────────────────────────────────────
create table if not exists users (
  id            bigint generated always as identity primary key,
  email         text unique not null,
  password_hash text not null,
  name          text,
  surname       text,
  age           integer,
  gender        text,
  created_at    timestamptz default now()
);


-- ────────────────────────────────────────────────────────────────
-- 2. JOBS TABLE
--    All job listings. id is uuid for easier external references.
-- ────────────────────────────────────────────────────────────────
create table if not exists jobs (
  id                   uuid primary key default gen_random_uuid(),

  -- Display
  emoji                text default '💼',
  title_en             text not null,
  title_bm             text,
  tagline_en           text,
  tagline_bm           text,

  -- Job ladder
  entry_job_en         text,
  entry_job_bm         text,
  target_job_en        text,
  target_job_bm        text,
  entry_salary_min     integer default 1400,
  entry_salary_max     integer default 1800,
  target_salary_min    integer default 2500,
  target_salary_max    integer default 4000,
  timeline_years       text default '2–3',

  -- Matching / scoring metadata
  category             text not null,
  skills_affinity      text[],
  quiz_affinity        text[],
  poverty_exit_score   integer default 5,
  income_type          text default 'fixed',
  physically_demanding boolean default false,
  internet_required    boolean default false,
  cert_required        boolean default false,
  education_min        text default 'primary',
  min_age              integer default 16,

  -- Explore page display
  description_en       text,
  description_bm       text,
  duties_en            text,
  duties_bm            text,
  requirements_en      text[],
  requirements_bm      text[],
  perks_en             text[],
  perks_bm             text[],
  work_type            text default 'full-time',
  location_type        text default 'on-site',

  -- Roadmap phases (JSONB)
  phase1               jsonb,
  phase2               jsonb,
  phase3               jsonb,
  courses              jsonb,
  income_ladder        jsonb,

  -- Jobstreet integration
  jobstreet_keywords   text[],

  -- Localised strings
  safety_rating_en     text default 'Steady Growth',
  safety_rating_bm     text default 'Pertumbuhan Stabil',
  motivation_en        text,
  motivation_bm        text,

  created_at           timestamptz default now()
);

create index if not exists idx_jobs_category  on jobs(category);
create index if not exists idx_jobs_work_type on jobs(work_type);

alter table jobs enable row level security;

drop policy if exists "Public read-only jobs" on jobs;
create policy "Public read-only jobs"
  on jobs for select using (true);


-- ────────────────────────────────────────────────────────────────
-- 3. QUIZ RESULTS TABLE
--    user_id is bigint to match users.id
--    Removed: q_digital, q_ambition (removed from quiz)
-- ────────────────────────────────────────────────────────────────
create table if not exists quiz_results (
  id              bigint generated always as identity primary key,
  user_id         bigint not null references users(id) on delete cascade,

  -- Card / chip questions
  interest        text,
  income_mode     text,
  education       text,
  skills          text[],

  -- Slider questions (0=strongly disagree … 4=strongly agree)
  -- NOTE: q_digital and q_ambition removed from quiz per product decision
  q_people        integer,
  q_outdoor       integer,
  q_creative      integer,
  q_flexible      integer,
  q_growth        integer,
  q_leadership    integer,

  -- Metadata
  profile_summary jsonb,
  created_at      timestamptz default now(),
  updated_at      timestamptz default now(),

  unique(user_id)
);

create index if not exists idx_quiz_results_user_id on quiz_results(user_id);

alter table quiz_results enable row level security;

drop policy if exists "Users manage their own quiz results" on quiz_results;
create policy "Users manage their own quiz results"
  on quiz_results for all using (true);
  -- RLS via session is handled in Flask; anon key only used server-side.


-- Auto-update updated_at
create or replace function update_updated_at()
returns trigger language plpgsql as $$
begin new.updated_at = now(); return new; end;
$$;

drop trigger if exists quiz_results_updated_at on quiz_results;
create trigger quiz_results_updated_at
  before update on quiz_results
  for each row execute function update_updated_at();


-- ────────────────────────────────────────────────────────────────
-- 4. SAVED JOBS TABLE
--    user_id → bigint  (matches users.id)
--    job_id  → uuid    (matches jobs.id)
-- ────────────────────────────────────────────────────────────────
create table if not exists saved_jobs (
  id         bigint generated always as identity primary key,
  user_id    bigint not null references users(id) on delete cascade,
  job_id     uuid   not null references jobs(id)  on delete cascade,
  saved_at   timestamptz default now(),
  unique(user_id, job_id)
);

create index if not exists idx_saved_jobs_user_id on saved_jobs(user_id);
create index if not exists idx_saved_jobs_job_id  on saved_jobs(job_id);

alter table saved_jobs enable row level security;

drop policy if exists "Users manage their own saved jobs" on saved_jobs;
create policy "Users manage their own saved jobs"
  on saved_jobs for all using (true);


-- ────────────────────────────────────────────────────────────────
-- 5. CAREER PATHS TABLE (static AI fallback — unchanged)
-- ────────────────────────────────────────────────────────────────
create table if not exists career_paths (
  id         serial primary key,
  interest   text not null,
  lang       text not null,
  path_data  jsonb not null
);

create index if not exists idx_career_paths_interest_lang
  on career_paths (interest, lang);

alter table career_paths enable row level security;

drop policy if exists "Public read-only career paths" on career_paths;
create policy "Public read-only career paths"
  on career_paths for select using (true);
